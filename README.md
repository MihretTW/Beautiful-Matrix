#include <iostream>
#include<cmath>
using namespace std;
int main(){
    int n,c ,r;
    int row_index,colum_index;
    for (int i=1; i<=5;i++){
        for (int j=1;j<=5;j++){
            cin>>n;
            if(n==1){
            row_index=i;
            colum_index=j;}}}
                
    r=abs(row_index-3);
    c=abs(colum_index-3);
        cout<<r+c;
}
